#include <stdio.h>
#include <stdlib.h>


int compareAsc(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int compareDesc(const void *a, const void *b) {
    return (*(int *)b - *(int *)a);
}

char* twoArrays(int k, int n, int A[], int B[]) {
    
    qsort(A, n, sizeof(int), compareAsc);

    
    qsort(B, n, sizeof(int), compareDesc);

        for (int i = 0; i < n; i++) {
        if (A[i] + B[i] < k) {
            return "NO";
        }
    }

    return "YES";
}

int main() {
    int q;
    printf(" number of queries: ");
    scanf("%d", &q);

    while (q--) {
        int n, k;
        printf(" size of arrays and value of k: ");
        scanf("%d %d", &n, &k);

        int A[n], B[n];

        printf("array A: ");
        for (int i = 0; i < n; i++) {
            scanf("%d", &A[i]);
        }

        printf(" array B: ");
        for (int i = 0; i < n; i++) {
            scanf("%d", &B[i]);
        }

        
        printf("%s\n", twoArrays(k, n, A, B));
    }

    return 0;
}
