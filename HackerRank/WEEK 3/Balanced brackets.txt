#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char* isBalanced(char* s) {
    int n = strlen(s);
    char *stack = (char*)malloc(n * sizeof(char));
    int top = -1;

    for (int i = 0; i < n; i++) {
        char c = s[i];

        if (c == '(' || c == '[' || c == '{') {
            stack[++top] = c;
        } else {
            if (top == -1) { 
                free(stack);
                static char no[] = "NO";
                return no;
            }
            char topChar = stack[top--];
            if ((c == ')' && topChar != '(') ||
                (c == ']' && topChar != '[') ||
                (c == '}' && topChar != '{')) {
                free(stack);
                static char no[] = "NO";
                return no;
            }
        }
    }

    char *result;
    if (top == -1) {
        static char yes[] = "YES";
        result = yes;
    } else {
        static char no[] = "NO";
        result = no;
    }

    free(stack);
    return result;
}

int main() {
    int n;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        char s[1000];
        scanf("%s", s);
        printf("%s\n", isBalanced(s));
    }

    return 0;
}
